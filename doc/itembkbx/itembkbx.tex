書式です。

\begin{boxnote}
\begin{verbatim}
\begin{breakitembox}<key=val>[配置]{見出し}
\begin{breakitembbox}[配置]{見出し}
\begin{breakitemtbbox}<上辺見出しの配置>[下辺見出しの配置]{上辺見出し}{下辺見出し}

key は
    parindent
    hsep
配置は
  l : 左
  c : 中央
  r : 右
そのデフォルトは
  breakitembox   : 中央
  breakitembbox  : 右
  breakitemtbbox : 上辺は左，下辺は右
\end{verbatim}
\end{boxnote}
