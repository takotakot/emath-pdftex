			\rightto<n>(5zw)"""$R>0$"
			\upto(1zw)
			\opbox{
				\begin{minipage}{4zw}
					\vspace{-\baselineskip}
					\begin{align*}
						A &\longleftarrow B\\
						B &\longleftarrow R
					\end{align*}
				\end{minipage}}
			\upto<n>`loop0'
			\leftto`loop0'
		\end{migibunki}
		\begin{sitabunki}
			\downto"$R=0$"
			\iobox{B を表示}
			\downto
			\termbox{終わり}
		\end{sitabunki}
	\end{ifbox}
\end{nagarezu}
