\section{接線}
\begin{mon}
  放物線 $y=x^2$ 上の点 $(-1,~1)$ における接線の方程式を求めよ．
\kaitou{$y=-2x-1$}
\end{mon}

\begin{rensyuu}
  放物線 $y=x^2$ に点 $(1,~0)$ から引いた接線の方程式を求めよ．
\kaitou{
  $y=x^2$ を微分して，$y'=2x$

  接点の $x$座標を $t$ とすれば接線の方程式は\
  \begin{equation}
    y-t^2=2t(x-t) \label{eq:setu1}
  \end{equation}
  これが点 $(1,~0)$ を通るから，
  \[0-t^2=2t(1-t)\]
  これを解いて， $t=0,~2$
  
  それぞれ \eqref{eq:setu1} に代入して，求める方程式は
  \[ \boldsymbol{y=0,~y=4x-4} \]
}
\end{rensyuu}
